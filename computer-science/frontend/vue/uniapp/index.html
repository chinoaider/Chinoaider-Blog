server.js

cat > server.js <<'EOF'
const WebSocket = require('ws');
const os = require('os');

function getIP() {
  const faces = os.networkInterfaces();
  for (const name of Object.keys(faces)) {
    for (const face of faces[name]) {
      if (face.family === 'IPv4' && !face.internal && face.address.startsWith('192.168')) {
        return face.address;          // 只要 192.168 段
      }
    }
  }
  return 'localhost';
}

const wss = new WebSocket.Server({ port: 8080 });
wss.on('connection', ws => {
  console.log('>>> 有人连上了，当前人数：', wss.clients.size);
  ws.on('message', buf => {
    const msg = buf.toString();
    wss.clients.forEach(c => c.readyState === 1 && c.send(msg));
  });
  ws.on('close', () => console.log('>>> 有人退出，剩余：', wss.clients.size));
});

console.log(`WebSocket 已启动: ws://${getIP()}:8080`);
EOF
